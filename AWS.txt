pipeline {
    agent any
    tools{
        maven 'Maven-3.9'
    }
    stages {
        stage('git repo & clean') {
            steps {
                bat "rmdir  /s /q maven-java-sample"
                bat "git clone https://github.com/sushmitha579/maven-java-sample.git"
                bat "mvn clean -f maven-java-sample"
            }
        }
        stage('install') {
            steps {
                bat "mvn install -f maven-java-sample"
            }
        }
        stage('test') {
            steps {
                bat "mvn test -f maven-java-sample"
            }
        }
        stage('package') {
            steps {
                bat "mvn package -f maven-java-sample"
            }
        }
    }
}
